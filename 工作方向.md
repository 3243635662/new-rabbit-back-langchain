### 📤 3. 生产者：Controller 接收文件 → 推入队列
```typescript
import { Controller, Post, UseInterceptors, UploadedFile, Inject, Req } from '@nestjs/common';
import { FileInterceptor } from '@nestjs/platform-express';
import { diskStorage } from 'multer';
import { InjectQueue } from '@nestjs/bullmq';
import { Queue } from 'bullmq';
import { v4 as uuidv4 } from 'uuid';
import { join } from 'path';
import { mkdirSync } from 'fs';

@Controller('rag')
export class RagController {
  constructor(@InjectQueue('rag-queue') private readonly ragQueue: Queue) {}

  @Post('upload')
  @UseInterceptors(
    FileInterceptor('file', {
      // 临时落盘，供 LangChain 解析使用
      storage: diskStorage({
        destination: (_req, _file, cb) => {
          const dir = join(__dirname, `../../tmp/rag_${uuidv4()}`);
          mkdirSync(dir, { recursive: true });
          cb(null, dir);
        },
        filename: (_req, file, cb) => cb(null, file.originalname), // 保留后缀供 Loader 识别
      }),
      limits: { fileSize: 50 * 1024 * 1024 }, // 50MB
    })
  )
  async uploadFile(
    @UploadedFile() file: Express.Multer.File,
    @Req() req: { user: { id: string } }
  ) {
    const taskId = uuidv4();
    
    // 推入队列（不阻塞请求，立刻返回）
    await this.ragQueue.add('process-document', {
      taskId,
      filePath: file.path,
      fileName: file.originalname,
      mimeType: file.mimetype,
      userId: req.user.id,
    }, {
      attempts: 3,          // 失败重试3次
      backoff: { type: 'exponential', delay: 5000 }, // 指数退避
      removeOnComplete: { age: 86400, count: 100 }, // 成功24h后自动清理
      removeOnFail: { age: 172800 },
    });

    return { code: 0, message: '已接收，处理中', taskId };
  }
}
```

### 🛠 4. 消费者：Processor 异步执行 LangChain RAG 流水线
```typescript
import { Processor, Process } from '@nestjs/bullmq';
import { Job } from 'bullmq';
import { Injectable, Logger } from '@nestjs/common';
import { PDFLoader } from '@langchain/community/document_loaders/fs/pdf';
import { DocxLoader } from '@langchain/community/document_loaders/fs/docx';
import { TextLoader } from '@langchain/community/document_loaders/fs/text';
import { RecursiveCharacterTextSplitter } from 'langchain/text_splitter';
import * as fs from 'fs/promises';
import * as path from 'path';

@Injectable()
@Processor('rag-queue')
export class RagProcessor {
  private readonly logger = new Logger(RagProcessor.name);

  @Process('process-document')
  async handle(job: Job<{ taskId: string; filePath: string; fileName: string; mimeType: string; userId: string }>) {
    const { taskId, filePath, fileName, mimeType, userId } = job.data;
    const tempDir = path.dirname(filePath);

    try {
      // 1️⃣ 更新进度：开始解析
      await job.updateProgress(10);

      // 2️⃣ 根据 MIME 类型选择 LangChain Loader
      let loader;
      if (mimeType.includes('pdf')) {
        loader = new PDFLoader(filePath);
      } else if (mimeType.includes('docx') || mimeType.includes('wordprocessingml')) {
        loader = new DocxLoader(filePath);
      } else if (mimeType.includes('text') || mimeType.includes('json')) {
        loader = new TextLoader(filePath);
      } else {
        throw new Error(`不支持的文件类型: ${mimeType}`);
      }

      await job.updateProgress(30);
      const docs = await loader.load();
      this.logger.log(`[ ${taskId} ] 解析完成，共 ${docs.length} 个文档块`);

      // 3️⃣ 文本分块（RAG 核心）
      await job.updateProgress(50);
      const splitter = new RecursiveCharacterTextSplitter({
        chunkSize: 800,
        chunkOverlap: 100,
      });
      const chunks = await splitter.splitDocuments(docs);

      // 4️⃣ ⚠️ 此处接入你的向量库（示例：假设为向量化+存储）
      await job.updateProgress(70);
      // const embeddings = await this.vectorStore.addDocuments(chunks, { userId, taskId });

      // 5️⃣ 上传原始文件到七牛（可选，用于溯源/前端下载）
      await job.updateProgress(85);
      // const qiniuKey = await this.qiniuService.upload(filePath, `rag/${userId}/${fileName}`);

      await job.updateProgress(100);
      this.logger.log(`[ ${taskId} ] ✅ RAG处理完成`);

    } catch (error) {
      this.logger.error(`[ ${taskId} ] ❌ 处理失败: ${error.message}`);
      throw error; // BullMQ 捕获后会自动重试
    } finally {
      // 🧹 必须清理本地临时文件（无论成功失败）
      try {
        await fs.rm(tempDir, { recursive: true, force: true });
        this.logger.log(`[ ${taskId} ] 🗑️ 临时目录已清理`);
      } catch (e) {
        this.logger.warn(`清理临时文件失败: ${e.message}`);
      }
    }
  }
}
```

### 🔍 5. 客户端如何查询进度？（配套接口）
```typescript
import { Controller, Get, Param, Inject } from '@nestjs/common';
import { InjectQueue } from '@nestjs/bullmq';
import { Queue } from 'bullmq';

@Controller('rag')
export class RagController {
  constructor(@InjectQueue('rag-queue') private readonly ragQueue: Queue) {}

  @Get('status/:taskId')
  async getTaskStatus(@Param('taskId') taskId: string) {
    const job = await this.ragQueue.getJob(taskId);
    if (!job) return { code: 1, message: '任务不存在' };

    const state = await job.getState(); // 'waiting' | 'active' | 'completed' | 'failed'
    const progress = job.progress || 0;
    const failedReason = state === 'failed' ? job.failedReason : null;

    return {
      code: 0,
      taskId,
      status: state,
      progress: `${progress}%`,
      error: failedReason,
    };
  }
}
```

### 🚨 生产环境必看的 5 个坑
| 坑 | 现象 | 解法 |
|:---|:---|:---|
| **1. 队列堆积不消费** | 任务卡在 `waiting` | 检查 Redis 连接、Processor 是否注册、NestJS 是否启动 `RagModule` |
| **2. 内存泄漏 OOM** | LangChain 解析大 PDF 时进程崩溃 | 限制 `chunkSize`，或改用流式解析器；Node 启动加 `--max-old-space-size=4096` |
| **3. 临时文件删不掉** | 磁盘被占满 | 必须用 `try...finally` + `fs.rm(..., { recursive: true })` |
| **4. 重试风暴** | 失败任务无限重试拖垮 Redis | 配置 `attempts: 3` + `backoff`，失败状态记录到 DB 供人工介入 |
| **5. 七牛上传慢** | 阻塞 Worker 线程 | 七牛上传必须走流式 `fs.createReadStream().pipe()`，不要用 `readFileSync` |
